Program to print the sum of all prime numbers between 1 to limit
echo "Enter the limit:"
read n
m=$n
sum=0
for ((i=1; i<n; i++))
  do
    flag=0
     for ((j=2; j<i; j++))
      do
       if [ $((i%j)) -eq 0 ]
        then 
        flag=1
       fi
      done
    if [ $flag -eq 0 ]
     then
      sum=$(($sum+$i))
    fi
  done
  echo "Sum of the non prime Numbers upto $m is $sum"
